cmake_minimum_required(VERSION 3.14)
project(USBMonitorClient VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

if(MINGW)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -static-libgcc -static-libstdc++")
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -static")
endif()

add_executable(usbmonitor_client
    main.cpp
    client_app.cpp
    schannel_tls.cpp
    ${CMAKE_SOURCE_DIR}/../protocol/protocol.cpp
)

target_link_libraries(usbmonitor_client PRIVATE
    ws2_32
    secur32
    crypt32
)

target_include_directories(usbmonitor_client PRIVATE
    ${CMAKE_SOURCE_DIR}/..
)
